# 💡 Примеры плагинов

## Пример 1: Простой плагин

Простейший плагин с одной командой.

### Файлы

```
src/plugins/hello/
├── __init__.py
├── plugin.py
└── config.yaml
```

### plugin.py

```python
from src.core.plugin import Plugin


class HelloPlugin(Plugin):
    name = "hello"
    version = "1.0.0"

    async def on_load(self) -> None:
        pass

    async def on_unload(self) -> None:
        pass

    def get_commands(self) -> list[dict]:
        return [
            {"name": "привет", "handler": self.say_hello, "config": self.get_command_config("привет")},
        ]

    async def say_hello(self, cmd):
        await cmd.reply(f"Привет, {cmd.user.name}!")
```

### config.yaml

```yaml
enabled: true

commands:
  привет:
    cooldown: 0
    enabled: true
```

## Пример 2: Плагин с правами

Команда доступна только модерам.

### plugin.py

```python
from src.core.plugin import Plugin


class AdminPlugin(Plugin):
    name = "admin"
    version = "1.0.0"

    async def on_load(self) -> None:
        pass

    async def on_unload(self) -> None:
        pass

    def get_commands(self) -> list[dict]:
        return [
            {"name": "бан", "handler": self.ban, "config": self.get_command_config("бан")},
        ]

    async def ban(self, cmd):
        # Проверка прав
        if not self.check_permission(cmd.user.name):
            await cmd.reply("У тебя нет прав!")
            return

        if not cmd.parameter:
            await cmd.reply("Укажи пользователя!")
            return

        await cmd.reply(f"Пользователь {cmd.parameter} забанен!")
```

### config.yaml

```yaml
enabled: true

commands:
  бан:
    cooldown: 0
    enabled: true
    requires_permission: true
```

## Пример 3: Плагин с API

Использует внешний API.

### plugin.py

```python
from src.core.plugin import Plugin


class WeatherPlugin(Plugin):
    name = "weather"
    version = "1.0.0"

    async def on_load(self) -> None:
        pass

    async def on_unload(self) -> None:
        pass

    def get_commands(self) -> list[dict]:
        return [
            {"name": "погода", "handler": self.weather, "config": self.get_command_config("погода")},
        ]

    async def weather(self, cmd):
        if not cmd.parameter:
            await cmd.reply("Укажи город: !погода Киев")
            return

        # Используем встроенный API
        result = await self.client.request("weather", cmd.parameter)
        await cmd.reply(result)
```

### config.yaml

```yaml
enabled: true

commands:
  погода:
    cooldown: 10
    enabled: true
```

## Пример 4: Плагин с кастомными настройками

### plugin.py

```python
from random import choice
from src.core.plugin import Plugin


class FunPlugin(Plugin):
    name = "fun"
    version = "1.0.0"

    async def on_load(self) -> None:
        pass

    async def on_unload(self) -> None:
        pass

    def get_commands(self) -> list[dict]:
        return [
            {"name": "зона", "handler": self.zone, "config": self.get_command_config("зона")},
        ]

    async def zone(self, cmd):
        # Получаем настройки из YAML
        words = self.settings.get("words", ["тихоня"])
        await cmd.reply(choice(words))
```

### config.yaml

```yaml
enabled: true

commands:
  зона:
    cooldown: 10
    enabled: true

settings:
  words:
    - "тихоня"
    - "лох"
    - "мент"
    - "шестерка"
    - "авторитет"
```

## Пример 5: Плагин с несколькими командами

### plugin.py

```python
from random import randint, choice
from src.core.plugin import Plugin


class GamesPlugin(Plugin):
    name = "games"
    version = "1.0.0"

    async def on_load(self) -> None:
        pass

    async def on_unload(self) -> None:
        pass

    def get_commands(self) -> list[dict]:
        return [
            {"name": "кости", "handler": self.dice, "config": self.get_command_config("кости")},
            {"name": "орёл_решка", "handler": self.coin, "config": self.get_command_config("орёл_решка")},
            {"name": "угадай", "handler": self.guess, "config": self.get_command_config("угадай")},
        ]

    async def dice(self, cmd):
        await cmd.reply(f"Выпало: {randint(1, 6)}")

    async def coin(self, cmd):
        await cmd.reply(choice(["Орёл", "Решка"]))

    async def guess(self, cmd):
        number = randint(1, 10)
        await cmd.reply(f"Я загадал число от 1 до 10. Угадай!")
```

### config.yaml

```yaml
enabled: true

commands:
  кости:
    cooldown: 5
    enabled: true
  орёл_решка:
    cooldown: 5
    enabled: true
  угадай:
    cooldown: 10
    enabled: true
```

## Полезные советы

1. **Всегда проверяй права** для модераторских команд
2. **Используй cooldown** чтобы избежать спама
3. **Обрабатывай пустые параметры** - не все пользователи будут их указывать
4. **Используй settings** для настраиваемых значений
